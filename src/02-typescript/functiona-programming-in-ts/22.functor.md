# Functor

https://www.youtube.com/watch?v=N4mft_n06Wo

## Идея

`Functor` позволяет поднять функцию `(A) => B` до работы со значением в контексте:

```ts
(A) => B
// превращается в
(F<A>) => F<B>
```

Если у типа есть `map`, мы можем преобразовать внутреннее значение, не меняя внешний контейнер.

- `Array<A>` -> `Array<B>`
- `Option<A>` -> `Option<B>`
- `Either<E, A>` -> `Either<E, B>`
- `List<A>` -> `List<B>`

## Законы

У `Functor` есть два закона:

```ts
map((x) => x) === identity;
map(compose(g, f)) === compose(map(g), map(f));
```

- `identity`: `map(x => x)` не должен ничего менять
- `composition`: `map(compose(g, f))` должен давать тот же результат, что и `compose(map(g), map(f))`

## Перед demo

- `Either<E, A>` обычно мапится только по `Right<A>`, а `Left<E>` остаётся без изменений
- в TypeScript полностью общий `Functor<F>` быстро упирается в ограничения `HKT`

## Demo

```ts
type Option<A> = Some<A> | None;

export interface Some<A> {
  _tag: "Some";
  value: A;
}

export interface None {
  _tag: "None";
}
export const some = <A>(x: A): Option<A> => ({ _tag: "Some", value: x });
export const none = <A>(x: A): Option<A> => ({ _tag: "None" });
export const isNone = <A>(x: Option<A>): x is None => x._tag === "None";

type Match = <A, B>(
  onNone: () => B,
  onSome: (a: A) => B,
) => (x: Option<A>) => B;

const match: Match = (onNone, onSome) => (x) =>
  isNone(x) ? onNone() : onSome(x.value);
```

```ts
type StrLength = (x: string) => number;
const strLength: StrLength = (x) => x.length;

console.log(strLength("abcd")); // 4

type OptionStrLength = (Fx: Option<string>) => Option<number>;
const strLength1: OptionStrLength = (Fx) =>
  match(
    () => none,
    (v: string) => some(strLength(value)),
  )(Fx);

console.log(strLength1(some("abcd"))); // {_tag: "Some", value: 4}
console.log(strLength1(none)); // {_tag: "None"}

// PointFree

const cstrLengthPointfree = match(
  () => none,
  (v: string) => some(strLength(v)),
);
```

```ts
type Increment = (x: number) => number;
const increment: Increment = (x) => x + 1;

type OptionIncrement = (Fx: Option<number>) => Option<number>;
const increment1: OptionIncrement = (Fx) =>
  match(
    () => none,
    (v) => some(increment(v)),
  )(Fx);

// PointFree

const incrementPointFree: OptionIncrement = match(
  () => none,
  (v) => some(increment(v)),
);

console.log(incrementPointFree(some(12))); // {_tag: "Some", value: 13}
console.log(incrementPointFree(none)); // {_tag: " "}
```

```ts
type MapOption = <A, B>(f: (x: A) => B) => (Fx: Option<A>) => Option<B>;

const map_option: MapOption = (f) => (Fx) =>
  match(
    () => none,
    (v: Parameters<typeof f>[0]) => some(f(v)),
  )(Fx);

// Point free

const map_option_point_free: MapOption = (f) =>
  match(
    () => none,
    (v) => some(f(v)),
  );
```

```ts
const strLength2 = map_option_point_free(strLength);
```

```ts
const list1: List<string> = cons("a", cons("b", cons(c, nil)));

type ListStrLength = (x: List<string>) => List<number>;
type ListIncrement = (x: List<number>) => List<number>;

// We don't want to implement this function for list, we want to be able to reuse as much as possible

type MapList = <A, B>(f: (x: A) => B) => (Fx: List<A>) => List<B>;
const map_list: MapList = (f) =>
  matchList(
    () => nil,
    (head, tail) => cons(f(head), map_list(f)(tail)),

const strLength3 = map_list(strLength);
console.log(showList(strLength3(list1))); // 1, 1, 1;

const increment3 = map_list(increment);
console.log(showList(increment3(cons(1, cons(2, cons(3, nil)))))); // 2, 3, 4
```

```ts
type MapEither = <A, B, E>(
  f: (x: A) => B,
) => (Fx: Either<E, A>) => Either<E, B>;

const map_either: MapEither = (f) =>
  matchEither(
    (e) => left(e),
    (a) => right(f(a)),
  );

const increment4 = map_either(icrement);

console.log(increment4(12)); // {_tag: "Right", right: 13}
console.log(increment4(left("Error"))); // {_tag: "Left", left: "Error"}

const strLength4 = map_either(strLength);
console.log(strLength4(right("abcd"))); // {_tag: "Right", right: 4}
console.log(strLength4(left("Error"))); // {_tag: "Left", left: "Error"}
```
