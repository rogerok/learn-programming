# Type Class in fp-ts

https://www.youtube.com/watch?v=8llcVd9UVBQ

```ts
import {} from "fp-ts";

declare module "fp-ts/HKT" {
  interface URItoKind<A> {
    List: List<A>;
    Option: Option<A>;
  }

  // integer in name of interface is amount of parameters
  interface URItoKind2<E, A> {
    Either: Either<E, A>;
    Option: Option<A>;
  }
}
```

How does this work? Are we overriting `URItoKind` interface inside `HKT` module?

Let explain it by example

```ts
interface Person {
  firstname: string;
}

interface Person {
  lastname: string;
}

const john: Person = {
  firstname: "John",
  lastname: "Doe",
};
```

Interfaces with same name are merged by TS.

fp-ts uses module augmentation to add code to module inside fp-ts lib.
it gives possibility for developers to dynamically add their structures to `URItoKind` interfaces
This way developers can add their own type constructors fo fp-ts

this is not valid ts syntax,

```ts
interface Functor<F> {
  map: <A, B>(f: (x: A) => B) => (fa: F<A>) => F<B>; //
}
```

but we can fix it by fp-ts HKT

```ts
import { HKT } from "fp-ts/HKT";

interface Functor<F> {
  map: <A, B>(f: (x: A) => B) => (fa: HKT<F, A>) => HKT<F, B>;
}
```

But we also need to write boilerplate code for fp-ts type classes

```ts
import { HKT, Kind, URIS, Kind2, URIS2 } from "fp-ts/HKT";

interface Functor<F extends URIS> {
  URIL: F;
  map: <A, B>(f: (x: A) => B) => (fa: HKT<F, A>) => HKT<F, B>;
}

interface Functor1<F> {
  URI: F;
  map: <A, B>(f: (x: A) => B) => (fa: Kind<F, A>) => Kind<F, B>;
}

interface Functor2<F extends URIS2> {
  URI: F;
  map: <E, A, B>(f: (x: A) => B) => (fa: Kind2<F, E, A>) => Kind2<F, E, B>;
}

const optionFunctor: Functor1<"Option"> = {
  URI: "Option",
  map: (f) =>
    optionMatch(
      () => none,
      (a) => some(f(a)),
    ),
};

const eitherFunctor: Functor2<"Either"> = {
  URI: "Either",
  map: (f) =>
    eitherMatch(
      (e) => left(e),
      (a) => right(f(a)),
    ),
};

function lift<F>(
  F: Functor<F>,
): <A, B>(f: (x: A) => B) => (fa: HKT<F, A>) => HKT<F, B> {
  return (f) => (fa) => F.map(f)(fa);
}

// We need to overload lift functions for working with diffrent kinds of structures

function lift<F extends URIS2>(
  F: Functor2<F>,
): <E, A, B>(f: (x: A) => B) => (fa: Kind2<F, E, A>) => Kind2<F, E, B>;

function lift<F extends URIS>(
  F: Functor1<F>,
): <A, B>(f: (x: A) => B) => (fa: Kind<F, A>) => Kind<F, B>;

function lift<F>(
  F: Functor<F>,
): <A, B>(f: (x: A) => B) => (fa: HKT<F, A>) => HKT<F, B>;
function lift<F>(
  F: Functor<F>,
): <A, B>(f: (x: A) => B) => (fa: HKT<F, A>) => HKT<F, B> {
  return (f) => (fa) => F.map(f)(fa);
}

type Increment = (x: number) => number;
const increment: Increment = (x) => x + 1;

const liftIncrement = lift(optionFunctor)(increment);

console.log(liftIncrement(some(12))); // {_tag: 'Some', value: 13}
console.log(liftIncrement(none)); // {_tag: 'None'}

const liftIncrement2 = lift(eitherFunctor)(increment);
console.log(liftIncrement2(right(12))); // {_tag: 'Right', right: 13}
```
