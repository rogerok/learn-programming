# Covariant and Contravariant Functors

https://www.youtube.com/watch?v=CWk1YlWXFME

## Demo

```ts
import { HKT, Kind, Kind2, URIS, URIS2 } from "fp-ts/HKT";

interface Contravaraint<F> {
  URI: F;
  contramap: <A, B>(f: (b: B) => A) => (fa: HKT<F, A>) => HKTF<F, B>;
}
interface Contravaraint1<F extends URIS> {
  URI: F;
  contramap: <A, B>(f: (b: B) => A) => (fa: Kind<F, A>) => Kind<F, B>;
}

interface Contravaraint2<F extends URIS2> {
  URI: F;
  contramap: <E, A, B>(f: (b: B) => A) => (fa: Kind<F, E, A>) => Kind<F, E, B>;
}

type Predicate<A> = (a: A) => boolean;

declare module "fp-ts/HKT" {
  interface URItoKind<A> {
    Predicate: Predicate<A>;
  }
}

const predicateContravariant: Contravariant1<"Predicate"> = {
  URI: "Predicate",
  contramap: (f) => (predicateA) => compose(predicateA, f),
};

type IsNumberEven = Predicate<number>;
const isNumberEven: IsNumberEven = (x) => x % 2 === 0;

type Length = (str: string) => number;
const length: Length = (str) => str.length;

type LiftedLength = (x: Predicate<number>) => Predicate<string>;
const liftedLength: LiftedLength = (predicate) =>
  predicateContravariant.contramap(length);

type IsStringEven = Predicate<string>;
const isStringEvent: IsStringEven = liftedLength(isNumberEven);

isStringEven("book"); // true
```
