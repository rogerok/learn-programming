# fp-ts Tutorial | Chapter 1: pipe and flow

https://www.youtube.com/watch?v=WsKEIFirdVc&list=PLUMXrUa_EuePN94nJ2hAui5nWDj8RO3lH&index=1

```ts
import { pipe } from "fp-ts/function";

const size = (s: string) => s.length;

pipe("hello", size); //5

const isAtLeast3 = (n: number) => n >= 3;

pipe(
  "hello",
  size, // 5
  isAtLeast3, // true
);

const trim = (s: string) => s.trim();

pipe(
  "hi ",
  trim, // 'hi'
  size, // 2,
  isAtLeast3, // false
);

const isValid = (s: string) => pipe(s, trim, size, isAtLeast3);
```

## Pipe with two functions

```ts
const pipe = <A, B, C>(a, A, f: (a: A) => B, g: (b: B) => C): C => g(f(a));
```

## Flow

`flow` function performs function composition

```ts
import { flow } from "fp-ts/function";

const size = (s: string) => s.length;
const isAtLeast3 = (n: number) => n >= 3;

const isLongEnough = flow(size, isAtLeast3); // (s: string) => boolean
```

```ts
const concat = (s1: string, s2: string) => s1 + s2;

const isValid = flow(concat, trim, size, isAtLeast3); // (s1: string, s2: string) => boolean

const flow =
  <A, B, C>(f: (a: A) => B, g: (b: B) => C) =>
  (a: A): C =>
    g(f(a));
```
