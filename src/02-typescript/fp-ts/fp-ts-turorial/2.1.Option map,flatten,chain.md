```ts
const head = <A>(as: ReadonlyArray<A>): O.Option<A> =>
  as.length === 0 ? O.none : O.some(as[0]);

const getBestMovie = (titles: ReadonlyArray<string>): O.Option<string> =>
  pipe(
    titles,
    head, // Option<string>
  );
```

`O.map(f)` maps current value to a new value

```ts
pipe(o, O.map(f)); //

const getBestMovie = (titles: ReadonlyArray<string>): O.Option<string> =>
  pipe(
    titles,
    // ["An American in Rome", "Winter Holidays"]
    head, // Option<string>
    // O.some('An American in Rome')
    O.map((s) => s.toUpperCase()),
    // O.some('AN AMERICAN IN ROME')
    O.map((s) => `Best - ${s}`),
    // O.some('Best - AN AMERICAN IN ROME')
  );

getBestMovie(["An American in Rome", "Winter Holidays"]);
```

```ts
const toUpperCase = (s: string) => s.toUpperCase();
const addPrefix = (prefix: string) => (s: string) => prefix + s;

const getBestMovie = (titles: ReadonlyArray<string>): O.Option<string> =>
  pipe(titles, head, O.map(toUpperCase), O.map(addPrefix("Best - ")));
```

## Option flatten

```ts
const invereHead = (ns: ReadonlyArray<number>) =>
  pipe(
    ns,
    head, // Option<number>
    O.map(inverse), // Option<Option<number>>
  );

const invereHeadFlat = (ns: ReadonlyArray<number>) =>
  pipe(
    ns,
    head, // Option<number>
    O.chain(inverse), // Option<number>
  );
```
