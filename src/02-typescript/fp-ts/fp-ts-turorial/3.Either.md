```ts
import { makeMatch } from "ts-adt/MakeADT";

const matchError = makeMatch("type");

type AccountFrozenError = E.left({
  type: 'AccontFrozen',
  message: 'Frozen account'
})

const checkout = (cart: Cart) => (account: Account) =>
  pipe(
    account,
    pay(cart.total),
    E.match(
      matchError({
        AccountFrozen: (e) => "account frozen",
      }),
      (a) => "Handle success",
    ),
  );
```
