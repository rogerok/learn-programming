# Option

We're using `Option` type to avoid throwing an error

https://www.youtube.com/watch?v=Xh05ynb-DAo

```ts
type Option<A> = Some<A> | None;
```

```ts
import * as O from "fp-ts/Option";

const inverse = (x: number): O.Option<number> =>
  x === 0 ? O.none : O.some(1 / x);

inverse(0); // O.none
inverse(2); // O.some(0.5)
```

## Matching options

We can use Option type for matching options

```ts
O.match(onNone, onSome);
O.fold(onNone, onSome);
```

```ts
const getUiMessageWithInverse = (x: number): string =>
  pipe(
    x,
    inverse, // Option<number>,
    O.match(
      () => `Cannot get the inverse of ${x}`,
      (ix) => `The inverse of ${x} is ${ix}`,
    ), // string,
  );

getUiMessagewithInverse(0); // "Cannot get the inverse of 0;
getUiMessagewithInverse(0); // "The inverse of 2 is 0.5;
```

```ts
import { identity } from "fp-ts/functions";
const safeInverse = (x: number): number =>
  pipe(
    x,
    inverse,
    O.match(
      () => 0,
      // (ix) => ix,
      identity,
    ), // number,
  );

const safeInverse2 = (x: number): number =>
  pipe(
    x,
    inverse,
    O.getOrElse(() => 0), // number
  );
```

`O.getOrElse` only valid, if the return same type, but if we need to extend returned type, we can use
`0.getOrElseW`

```ts
const f = (a: Option<number>) =>
  pipe(
    a,
    O.getOrElseW(() => "invalid"), // number | string,
  );
```

Same for `0.match`

```ts
const f = (a: Option<number>) => (
  pipe,
  a,
  O.matchW(
    () => 0,
    (n) => `The value is ${n}!`,
  ) // number | string
);
```

Optional values typically represented by `Nullable` type

```ts
type Nullable<A> = A | null | undefined;
```

To convert nullable, we can use `O.fromNullable(nullableValue)`
It will return `Some` or `None`

```ts
const value1: number | null = 3;
const value2: number | null = null;

O.fromNullable(value1); // O.some(3)
O.fromNullable(value2); // O.none
```
