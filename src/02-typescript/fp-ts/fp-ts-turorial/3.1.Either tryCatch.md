## E.tryCatch

```ts
const tryCatch: <E, A>(f: () => A, onThrow: (e: unknown) => E) => Either<E, A>;
```

```ts
const jsonParse = (json: string): E.Either<Error, unknown> =>
  E.tryCatch(() => JSON.parse(text), E.toError);
```

## E.tryCatchK

```ts
const tryCatchK: <A extends unknown[], B, E>(
  f: (...a: A) => B,
  onThrow: (e: unknown) => E,
) => (...a: A) => Either<E, B>;
```

```ts
const jsonParse = E.tryCatchK(JSON.parse, E.toError);
```

```ts
type JsonParseError = {
  type: "JsonParseError";
  error: Error;
};
const jsonParse: (text: string) => E.Either<JsonParseError, unknown> =
  E.tryCatchK(JSON.parse, (e) => ({
    type: "JsonParseError",
    error: E.toError(e),
  }));

jsonParse('{"foo":"bar"}'); // E.right({foo: 'bar'})

jsonParse("{invalid}");
```
