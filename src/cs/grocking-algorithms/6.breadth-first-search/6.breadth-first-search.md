const findMangoSeller = (graph, source) => {
  const queue = [source];
  const checked = new Set();

  while (queue.length) {
    const current = queue.shift();
    checked.add(current);

    if (current.includes("m")) {
      console.log(current + " is mango seller");
      break;
    }

    for (let neigbour of graph[current]) {
      if (!checked.has(neigbour)) {
        queue.push(neigbour);
      }
    }
  }
};
