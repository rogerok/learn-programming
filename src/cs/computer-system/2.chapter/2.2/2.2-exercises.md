## Упражнение 1: Таблица B2U и B2T для 4-битных чисел

**Сложность:** начальный

### Условие

Реализуй функцию `printTable()`, которая перебирает все 16 значений типа `u4` (от `0b0000` до `0b1111`) и для каждого выводит:

- двоичное представление (4 знака, с ведущими нулями)
- B2U_4 — значение как беззнаковое число
- B2T_4 — значение как знаковое (через `@bitCast`)

Ожидаемый вывод (частично):

```
0000: B2U=0,  B2T=0
0001: B2U=1,  B2T=1
...
0111: B2U=7,  B2T=7   <- TMax_4
1000: B2U=8,  B2T=-8  <- TMin_4
1011: B2U=11, B2T=-5
1111: B2U=15, B2T=-1
```

Затем добавь отдельные тесты через `std.testing.expectEqual`.

> [!tip]- Подсказка 1: синтаксис `while` с постфиксом
> В Zig цикл с шагом пишется так:
> ```zig
> var i: u4 = 0;
> while (true) : (i +%= 1) {
>     // тело
>     if (i == 15) break; // u4 не может стать 16, нужен ручной выход
> }
> ```
> `: (i +%= 1)` — это **постфиксный шаг**: выполняется после каждой итерации тела.
> `+%=` — оператор сложения с wrapping (обёртыванием), который не паникует при переполнении.

> [!tip]- Подсказка 2: `@bitCast` для `u4 -> i4`
> ```zig
> const bits: u4 = 0b1011;
> const as_signed: i4 = @bitCast(bits);
> // as_signed == -5
> ```
> `@bitCast` переинтерпретирует те же биты с другим типом. Размер (в битах) обеих сторон должен совпадать.

> [!tip]- Подсказка 3: вывод двоичного числа с ведущими нулями
> ```zig
> std.debug.print("{b:0>4}", .{val}); // "0011" для val=3
> ```
> Формат `{b:0>4}` означает: двоичный (`b`), дополнить нулями слева (`0>`), минимальная ширина 4.

> [!warning]- Решение
> ```zig
> const std = @import("std");
>
> fn printTable() void {
>     std.debug.print("{s:<6} {s:<8} {s}\n", .{ "Bits", "B2U", "B2T" });
>     std.debug.print("{s}\n", .{"----  ------  ----"});
>
>     var i: u4 = 0;
>     while (true) : (i +%= 1) {
>         const as_signed: i4 = @bitCast(i);
>         std.debug.print("{b:0>4}  B2U={d:<3}   B2T={d}\n", .{ i, i, as_signed });
>         if (i == 15) break;
>     }
> }
>
> test "B2T_4 особые значения" {
>     // TMin_4: биты 1000 -> -8
>     const tmin_bits: u4 = 0b1000;
>     const tmin: i4 = @bitCast(tmin_bits);
>     try std.testing.expectEqual(@as(i4, -8), tmin);
>
>     // TMax_4: биты 0111 -> 7
>     const tmax_bits: u4 = 0b0111;
>     const tmax: i4 = @bitCast(tmax_bits);
>     try std.testing.expectEqual(@as(i4, 7), tmax);
>
>     // Все единицы: 1111 -> -1
>     const all_ones: u4 = 0b1111;
>     const minus_one: i4 = @bitCast(all_ones);
>     try std.testing.expectEqual(@as(i4, -1), minus_one);
>
>     // 1011 -> -5
>     const bits_1011: u4 = 0b1011;
>     const val: i4 = @bitCast(bits_1011);
>     try std.testing.expectEqual(@as(i4, -5), val);
> }
>
> pub fn main() void {
>     printTable();
> }
> ```

### Разбор

Ключевые наблюдения из таблицы:
- При знаковом бите = 0 (строки `0000`–`0111`): B2U == B2T. Оба представления дают одинаковый результат.
- При знаковом бите = 1 (строки `1000`–`1111`): B2T = B2U − 16. Разница ровно `2^4 = 16`. Это и есть суть формулы T2U/U2T.
- `1111` — это одновременно `UMax_4 = 15` и `−1`. Один и тот же битовый паттерн, разные интерпретации.

## Упражнение 2: Функции T2U и U2T через формулу

**Сложность:** начальный

### Условие

Реализуй две функции для 16-битных чисел:

```zig
fn t2u16(x: i16) u16  // T2U_16: знаковое -> беззнаковое
fn u2t16(u: u16) i16  // U2T_16: беззнаковое -> знаковое
```

**Требование:** используй математические формулы из конспекта, а не `@bitCast` напрямую. Потом проверь, что результат совпадает с `@bitCast`.

Формулы (для w = 16):
```
T2U_16(x) = x + 65536,  если x < 0
             x,           если x >= 0

U2T_16(u) = u - 65536,   если u > 32767
             u,           если u <= 32767
```

> [!tip]- Подсказка 1: промежуточный тип для формулы
> Чтобы применить формулу `x + 65536` для `i16`, нужен промежуточный тип достаточной ширины:
> ```zig
> const x: i16 = -12345;
> const wide: i32 = x; // sign extension
> const result: u16 = @intCast(wide + 65536); // теперь >= 0, вмещается в u16
> ```

> [!tip]- Подсказка 2: константа `2^16`
> `65536 = 1 << 16`. Можно использовать `@as(i32, 1) << 16` или просто литерал `65536`.

> [!tip]- Подсказка 3: проверка совпадения с `@bitCast`
> ```zig
> const via_formula = t2u16(-12345);
> const via_bitcast: u16 = @bitCast(@as(i16, -12345));
> try std.testing.expectEqual(via_formula, via_bitcast);
> ```

> [!warning]- Решение
> ```zig
> const std = @import("std");
>
> fn t2u16(x: i16) u16 {
>     // T2U_16: если x < 0, добавляем 2^16
>     if (x < 0) {
>         // Расширяем до i32, чтобы не переполниться при сложении
>         const wide: i32 = x;
>         return @intCast(wide + 65536);
>     }
>     return @intCast(x);
> }
>
> fn u2t16(u: u16) i16 {
>     // U2T_16: если u > TMax_16 (32767), вычитаем 2^16
>     if (u > 32767) {
>         // Расширяем до i32, чтобы вычитание не переполнилось
>         const wide: i32 = u;
>         return @intCast(wide - 65536);
>     }
>     return @intCast(u);
> }
>
> test "T2U_16 базовые случаи" {
>     try std.testing.expectEqual(@as(u16, 53191), t2u16(-12345));
>     try std.testing.expectEqual(@as(u16, 65535), t2u16(-1));   // UMax_16
>     try std.testing.expectEqual(@as(u16, 0), t2u16(0));
>     try std.testing.expectEqual(@as(u16, 32767), t2u16(32767)); // TMax_16, без изменений
> }
>
> test "U2T_16 базовые случаи" {
>     try std.testing.expectEqual(@as(i16, -12345), u2t16(53191));
>     try std.testing.expectEqual(@as(i16, -1), u2t16(65535));    // UMax -> -1
>     try std.testing.expectEqual(@as(i16, 0), u2t16(0));
>     try std.testing.expectEqual(@as(i16, 32767), u2t16(32767)); // TMax, без изменений
> }
>
> test "совпадение формулы и @bitCast" {
>     // T2U через формулу == T2U через биты
>     const cases = [_]i16{ -12345, -1, -32768, 0, 1, 32767 };
>     for (cases) |x| {
>         const via_formula = t2u16(x);
>         const via_bitcast: u16 = @bitCast(x);
>         try std.testing.expectEqual(via_formula, via_bitcast);
>     }
>
>     // U2T через формулу == U2T через биты
>     const u_cases = [_]u16{ 53191, 65535, 32768, 0, 1, 32767 };
>     for (u_cases) |u| {
>         const via_formula = u2t16(u);
>         const via_bitcast: i16 = @bitCast(u);
>         try std.testing.expectEqual(via_formula, via_bitcast);
>     }
> }
> ```

### Разбор

Формулы T2U и U2T — это просто математическое описание того, что происходит с `@bitCast`. Если `x < 0`, то как беззнаковое оно "прыгает" на `2^w` вверх. Обратно: если беззнаковое > TMax, то как знаковое оно "прыгает" на `2^w` вниз.

Проверка совпадения с `@bitCast` подтверждает главный тезис главы: **T2U и U2T не меняют биты**, они меняют лишь интерпретацию.

## Упражнение 3: Опасные сравнения — Zig vs C

**Сложность:** начальный

### Условие

Реализуй функцию `analyzeComparisons()`, которая демонстрирует три ситуации, где C ведёт себя контринтуитивно из-за неявного приведения signed -> unsigned.

Для каждого случая:
1. Покажи, что C "думает" (воспроизведи через явный каст в Zig)
2. Покажи правильный результат
3. Объясни через комментарий — почему так

Случаи:
1. Сравнение `i32(-1)` и `u32(0)`: C говорит `-1 < 0U` = false
2. Обнаружение переполнения `i32::MIN - 1` через `@subWithOverflow`
3. Обнаружение переполнения `i32::MAX + 1` через `@addWithOverflow`

> [!tip]- Подсказка 1: имитация C-поведения в Zig
> Zig не допускает сравнение `i32` и `u32` напрямую. Чтобы воспроизвести поведение C, нужен явный каст:
> ```zig
> // C делает: (unsigned)(-1) < 0U
> const c_style = @as(u32, @bitCast(@as(i32, -1))) < @as(u32, 0);
> // c_style == false, потому что @bitCast(-1 as i32) = 4294967295
> ```

> [!tip]- Подсказка 2: `@subWithOverflow` и `@addWithOverflow`
> ```zig
> const result = @subWithOverflow(@as(i32, value), @as(i32, 1));
> // result[0] — результат операции (с wrapping)
> // result[1] — флаг переполнения: 1 если было переполнение, 0 если нет
> ```

> [!warning]- Решение
> ```zig
> const std = @import("std");
>
> fn analyzeComparisons() void {
>     // --- Случай 1: -1 vs 0U ---
>     const sv: i32 = -1;
>     const uv: u32 = 0;
>
>     // Правильное сравнение в Zig (оба к signed):
>     const correct = sv < @as(i32, @intCast(uv));
>     std.debug.print("Случай 1: i32(-1) < u32(0)?\n", .{});
>     std.debug.print("  Правильный ответ: {}\n", .{correct}); // true
>
>     // Как C: signed приводится к unsigned
>     const c_behavior = @as(u32, @bitCast(sv)) < uv;
>     std.debug.print("  C-поведение:      {}\n", .{c_behavior}); // false!
>     std.debug.print("  Причина: @bitCast(i32(-1)) = u32({})\n\n", .{@as(u32, @bitCast(sv))});
>
>     // --- Случай 2: TMin - 1 ---
>     const tmin: i32 = std.math.minInt(i32);
>     const sub_result = @subWithOverflow(tmin, @as(i32, 1));
>     std.debug.print("Случай 2: i32::MIN - 1\n", .{});
>     std.debug.print("  i32::MIN = {}\n", .{tmin});
>     std.debug.print("  Переполнение: {}\n", .{sub_result[1] == 1}); // true
>     std.debug.print("  Wrapping-результат: {} (это {}, не математически корректно)\n\n", .{
>         sub_result[0],
>         sub_result[0],
>     });
>
>     // --- Случай 3: TMax + 1 ---
>     const tmax: i32 = std.math.maxInt(i32);
>     const add_result = @addWithOverflow(tmax, @as(i32, 1));
>     std.debug.print("Случай 3: i32::MAX + 1\n", .{});
>     std.debug.print("  i32::MAX = {}\n", .{tmax});
>     std.debug.print("  Переполнение: {}\n", .{add_result[1] == 1}); // true
>     std.debug.print("  Wrapping-результат: {} (это TMin!)\n", .{add_result[0]});
> }
>
> test "опасное сравнение: C-поведение vs Zig" {
>     const sv: i32 = -1;
>     const uv: u32 = 0;
>
>     // Правильно: -1 < 0 (true)
>     try std.testing.expect(sv < @as(i32, @intCast(uv)));
>
>     // C-стиль: @bitCast(-1) = 4294967295, что НЕ меньше 0
>     try std.testing.expect(!(@as(u32, @bitCast(sv)) < uv));
> }
>
> test "обнаружение переполнения через @subWithOverflow" {
>     const tmin: i32 = std.math.minInt(i32);
>     const result = @subWithOverflow(tmin, @as(i32, 1));
>     try std.testing.expectEqual(@as(u1, 1), result[1]); // было переполнение
> }
>
> test "обнаружение переполнения через @addWithOverflow" {
>     const tmax: i32 = std.math.maxInt(i32);
>     const result = @addWithOverflow(tmax, @as(i32, 1));
>     try std.testing.expectEqual(@as(u1, 1), result[1]); // было переполнение
>     try std.testing.expectEqual(std.math.minInt(i32), result[0]); // TMax + 1 = TMin (wrapping)
> }
>
> pub fn main() void {
>     analyzeComparisons();
> }
> ```

### Разбор

Случай 1 — это знаменитое `-1 < 0U` из книги (таблица 2.12). C тихо делает T2U(-1) = UMax, и сравнение неожиданно ложное. Zig **не позволяет** написать это случайно — требует явного каста, и тем самым принуждает к осознанному выбору семантики.

`@subWithOverflow` / `@addWithOverflow` возвращают структуру `{результат, флаг}`. Это явная альтернатива C-шному UB при переполнении знакового числа.

## Упражнение 4: Sign extension и zero extension

**Сложность:** средний

### Условие

Реализуй функцию `showExtension(sx: i16)`, которая демонстрирует три вида расширения:

1. **Sign extension** `i16 → i32` — автоматически в Zig через присваивание
2. **Zero extension** `u16 → u32` — сначала `@bitCast` к `u16`, потом расширение
3. **C-style каст** `i16 → u32` — два шага: sign-ext до i32, потом `@bitCast` к u32

Для `sx = -12345` ожидаемый вывод:
```
i16(-12345) -> i32(-12345)    [hex: 0xFFFFCFC7] <- sign extension
u16(53191)  -> u32(53191)     [hex: 0x0000CFC7] <- zero extension
i16(-12345) -> u32(4294954951) [hex: 0xFFFFCFC7] <- C-style cast
```

> [!tip]- Подсказка 1: hex-вывод с ведущими нулями
> ```zig
> const bits: u32 = @bitCast(some_i32);
> std.debug.print("0x{x:0>8}", .{bits}); // "0xffff cfc7"
> ```
> Формат `{x:0>8}` — шестнадцатеричный, дополнить нулями слева до 8 символов.

> [!tip]- Подсказка 2: три шага для C-style каста
> В C `(unsigned)(short)(-12345)` означает:
> 1. `short` (-12345) → `int` (-12345) через sign extension
> 2. `int` (-12345) → `unsigned` (4294954951) через reinterpret
>
> В Zig это два явных действия:
> ```zig
> const step1: i32 = sx;         // sign extension (автоматически)
> const step2: u32 = @bitCast(step1); // reinterpret bits
> ```

> [!warning]- Решение
> ```zig
> const std = @import("std");
>
> fn showExtension(sx: i16) void {
>     // 1. Sign extension: i16 -> i32
>     // Zig делает это автоматически при присваивании к более широкому знаковому типу
>     const sign_ext: i32 = sx;
>     const sign_ext_bits: u32 = @bitCast(sign_ext);
>     std.debug.print("i16({d}) -> i32({d:<12}) [hex: 0x{x:0>8}] <- sign extension\n", .{
>         sx, sign_ext, sign_ext_bits,
>     });
>
>     // 2. Zero extension: сначала reinterpret i16 -> u16, потом u16 -> u32
>     const as_u16: u16 = @bitCast(sx);
>     const zero_ext: u32 = as_u16; // Zig расширяет u16 -> u32 нулями автоматически
>     std.debug.print("u16({d:<6}) -> u32({d:<12}) [hex: 0x{x:0>8}] <- zero extension\n", .{
>         as_u16, zero_ext, zero_ext,
>     });
>
>     // 3. C-style: i16 -> u32 (sign-ext до i32, потом reinterpret)
>     const c_style: u32 = @bitCast(sign_ext); // sign_ext уже i32
>     std.debug.print("i16({d}) -> u32({d:<12}) [hex: 0x{x:0>8}] <- C-style cast\n\n", .{
>         sx, c_style, c_style,
>     });
> }
>
> test "sign extension сохраняет значение" {
>     const sx: i16 = -12345;
>     const extended: i32 = sx;
>     try std.testing.expectEqual(@as(i32, -12345), extended);
>
>     // Hex: 0xFFFFCFC7
>     const bits: u32 = @bitCast(extended);
>     try std.testing.expectEqual(@as(u32, 0xFFFFCFC7), bits);
> }
>
> test "zero extension сохраняет значение" {
>     const sx: i16 = -12345;
>     const as_u16: u16 = @bitCast(sx);    // 53191 = 0xCFC7
>     const extended: u32 = as_u16;        // 0x0000CFC7
>     try std.testing.expectEqual(@as(u32, 0x0000CFC7), extended);
>     try std.testing.expectEqual(@as(u32, 53191), extended);
> }
>
> test "C-style i16->u32 = sign_ext + bitcast" {
>     const sx: i16 = -12345;
>     const step1: i32 = sx;
>     const c_style: u32 = @bitCast(step1);
>     try std.testing.expectEqual(@as(u32, 4294954951), c_style);
>     try std.testing.expectEqual(@as(u32, 0xFFFFCFC7), c_style);
> }
>
> test "граничный случай sx = -1" {
>     const sx: i16 = -1;
>
>     // sign ext: 0xFFFFFFFF = 4294967295 как u32
>     const sign_ext: i32 = sx;
>     const sign_bits: u32 = @bitCast(sign_ext);
>     try std.testing.expectEqual(@as(u32, 0xFFFFFFFF), sign_bits);
>
>     // zero ext: 0x0000FFFF = 65535
>     const as_u16: u16 = @bitCast(sx);
>     const zero_ext: u32 = as_u16;
>     try std.testing.expectEqual(@as(u32, 0x0000FFFF), zero_ext);
>
>     // c-style: 0xFFFFFFFF = 4294967295
>     const c_style: u32 = @bitCast(sign_ext);
>     try std.testing.expectEqual(@as(u32, 4294967295), c_style);
> }
>
> pub fn main() void {
>     showExtension(-12345);
>     showExtension(-1);
>     showExtension(1);
> }
> ```

### Разбор

Sign extension и zero extension дают разные результаты для одних и тех же битов, потому что zero extension забывает о знаке и просто добавляет нули, а sign extension размножает знаковый бит.

Ключевое наблюдение из тестов: для `-1` (биты `0xFFFF`):
- sign ext → `0xFFFFFFFF` (все единицы — `-1` в i32)
- zero ext → `0x0000FFFF` (65535 — просто 65535 в u32)

Это подтверждает: расширение — не арифметика, а **копирование битов** по правилу типа.

## Упражнение 5: Усечение — `@truncate` vs `@intCast`

**Сложность:** средний

### Условие

Реализуй функцию `showTruncation(x: i32)`, которая демонстрирует два инструмента усечения:

- `@truncate` к `i16` — просто отбрасывает старшие биты
- `@truncate` к `i8` — ещё агрессивнее
- `@intCast` к `i16` — безопасная версия: паника если не вмещается

Затем:
- Проверь числовые значения тестами
- Добавь тест, который **ловит** попытку небезопасного `@intCast` через предварительную проверку диапазона

> [!tip]- Подсказка 1: `@truncate` и знаковость
> `@truncate` работает с битами, игнорируя знак. Биты усекаются, потом применяется тип результата:
> ```zig
> const x: i32 = 53191; // 0x0000CFC7
> const t: i16 = @truncate(x); // берёт младшие 16 бит: 0xCFC7 -> B2T_16 = -12345
> ```
> Обрати внимание: `53191` как `i32` положительное, но его младшие 16 бит интерпретируются как `i16` знаковое.

> [!tip]- Подсказка 2: `std.math.cast` как безопасная альтернатива
> ```zig
> const result: ?i16 = std.math.cast(i16, x);
> if (result) |val| {
>     // вмещается
> } else {
>     // не вмещается — null вместо паники
> }
> ```

> [!tip]- Подсказка 3: почему 53191 не помещается в i16
> `i16` может хранить от -32768 до 32767. `53191 > 32767`, поэтому `@intCast(i32(53191)) -> i16` — паника.
> При этом `@truncate` просто берёт биты и получает -12345.

> [!warning]- Решение
> ```zig
> const std = @import("std");
>
> fn showTruncation(x: i32) void {
>     std.debug.print("i32({d}) [hex: 0x{x:0>8}]\n", .{ x, @as(u32, @bitCast(x)) });
>
>     // @truncate к i16: отбрасывает старшие 16 бит
>     const t16: i16 = @truncate(x);
>     std.debug.print("  @truncate -> i16({d})  [биты 0x{x:0>4}]\n", .{ t16, @as(u16, @bitCast(t16)) });
>
>     // @truncate к i8: отбрасывает старшие 24 бита
>     const t8: i8 = @truncate(x);
>     std.debug.print("  @truncate -> i8({d})   [биты 0x{x:0>2}]\n", .{ t8, @as(u8, @bitCast(t8)) });
>
>     // @intCast: безопасен только если значение вмещается
>     const safe = std.math.cast(i16, x);
>     if (safe) |val| {
>         std.debug.print("  @intCast  -> i16({d})  [вмещается]\n", .{val});
>     } else {
>         std.debug.print("  @intCast  -> ПАНИКА!    [i32({d}) не вмещается в i16]\n", .{x});
>     }
>     std.debug.print("\n", .{});
> }
>
> test "@truncate: 53191 -> i16 и i8" {
>     const x: i32 = 53191; // 0x0000CFC7
>
>     // Усечение к i16: 0xCFC7 как знаковое = -12345
>     const t16: i16 = @truncate(x);
>     try std.testing.expectEqual(@as(i16, -12345), t16);
>
>     // Усечение к i8: 0xC7 как знаковое
>     // 0xC7 = 199 как u8, но как i8: 199 - 256 = -57
>     const t8: i8 = @truncate(x);
>     try std.testing.expectEqual(@as(i8, -57), t8);
> }
>
> test "@truncate: 12345 -> i16 и i8" {
>     const x: i32 = 12345; // 0x00003039
>
>     // 12345 вмещается в i16 (< 32767)
>     const t16: i16 = @truncate(x);
>     try std.testing.expectEqual(@as(i16, 12345), t16);
>
>     // 12345 mod 256 = 57 (0x39), вмещается в i8
>     const t8: i8 = @truncate(x);
>     try std.testing.expectEqual(@as(i8, 57), t8);
> }
>
> test "@intCast безопасен для 12345, опасен для 53191" {
>     // 12345 вмещается в i16 — std.math.cast возвращает значение
>     const ok = std.math.cast(i16, @as(i32, 12345));
>     try std.testing.expect(ok != null);
>     try std.testing.expectEqual(@as(i16, 12345), ok.?);
>
>     // 53191 не вмещается в i16 — std.math.cast возвращает null
>     const fail = std.math.cast(i16, @as(i32, 53191));
>     try std.testing.expect(fail == null);
> }
>
> pub fn main() void {
>     showTruncation(53191);
>     showTruncation(12345);
>     showTruncation(-12345);
> }
> ```

### Разбор

`@truncate` — это операция над битами: взять `k` младших битов и переинтерпретировать с новым типом. Числовое значение при этом может кардинально измениться (`53191 -> -12345`).

`@intCast` — операция над **числовыми значениями**: проверяет, что значение вмещается в диапазон целевого типа. Если не вмещается — паника.

`std.math.cast` — "мягкая" версия `@intCast`, возвращает `?T` вместо паники. Это предпочтительный вариант для кода, где входные данные непредсказуемы.

Формула усечения: `B2U_k(x mod 2^k)` — это именно то, что делает `@truncate`.

## Упражнение 6: Ловушка usize — баг с вычитанием длин

**Сложность:** продвинутый

### Условие

Реализуй две версии функции, которая суммирует первые `n` элементов массива:

```zig
// БАГОВАЯ версия: использует условие с вычитанием usize
fn sumFirstBuggy(arr: []const u32, n: usize) u64

// ПРАВИЛЬНАЯ версия
fn sumFirstFixed(arr: []const u32, n: usize) u64
```

Баговая версия должна содержать паттерн `i <= n - 1` (или `arr.len - n`) в условии цикла — классическую ловушку из книги.

Дополнительно реализуй `isLonger` в двух вариантах:

```zig
fn isLongerBuggy(s: []const u8, t: []const u8) bool  // s.len - t.len > 0
fn isLongerFixed(s: []const u8, t: []const u8) bool  // прямое сравнение
```

> [!tip]- Подсказка 1: почему `n - 1` опасен при `n = 0`
> `usize` — беззнаковый тип. Для `usize`:
> ```
> 0 - 1 = usize.MAX  (например, 18446744073709551615 на 64-битной системе)
> ```
> Это **не** ошибка компиляции — это корректная арифметика беззнаковых чисел. Zig паникует в debug-режиме, но в `ReleaseFast` — нет.

> [!tip]- Подсказка 2: правильный паттерн для `sumFirst`
> Используй `@min(n, arr.len)` чтобы не выйти за границу, и `for` по срезу:
> ```zig
> const limit = @min(n, arr.len);
> for (arr[0..limit]) |val| { ... }
> ```

> [!tip]- Подсказка 3: как задокументировать баг без запуска
> В Zig нет `#[should_panic]` как в Rust, но можно:
> - Вынести баговую функцию в отдельный блок с комментарием `// НЕ ЗАПУСКАТЬ`
> - Или добавить проверку до потенциально опасной операции:
> ```zig
> // Явная проверка перед вычитанием usize
> if (n == 0) return 0; // guard
> ```

> [!warning]- Решение
> ```zig
> const std = @import("std");
>
> // БАГОВАЯ версия:
> // При n = 0: условие становится `i <= usize.MAX`, то есть ВСЕГДА true.
> // В debug-режиме Zig поймает underflow при n - 1, когда n = 0.
> // В ReleaseFast: бесконечный цикл с выходом за границы массива.
> fn sumFirstBuggy(arr: []const u32, n: usize) u64 {
>     var sum: u64 = 0;
>     var i: usize = 0;
>     // БАГ ЗДЕСЬ: при n = 0, выражение (n - 1) = usize.MAX
>     // Это паника в Debug, UB/бесконечный цикл в ReleaseFast
>     while (i <= n - 1) : (i += 1) {
>         sum += arr[i];
>     }
>     return sum;
> }
>
> // ПРАВИЛЬНАЯ версия:
> fn sumFirstFixed(arr: []const u32, n: usize) u64 {
>     var sum: u64 = 0;
>     // @min гарантирует, что не выйдем за границу arr
>     const limit = @min(n, arr.len);
>     for (arr[0..limit]) |val| {
>         sum += val;
>     }
>     return sum;
> }
>
> // БАГОВАЯ версия isLonger:
> // При s.len < t.len: s.len - t.len = usize underflow -> паника в Debug
> fn isLongerBuggy(s: []const u8, t: []const u8) bool {
>     // БАГ: если s короче t, usize wraparound даёт огромное число > 0
>     return s.len - t.len > 0;
> }
>
> // ПРАВИЛЬНАЯ версия:
> fn isLongerFixed(s: []const u8, t: []const u8) bool {
>     return s.len > t.len;
> }
>
> test "sumFirstFixed: обычные случаи" {
>     const arr = [_]u32{ 1, 2, 3, 4, 5 };
>     try std.testing.expectEqual(@as(u64, 6), sumFirstFixed(&arr, 3));   // 1+2+3
>     try std.testing.expectEqual(@as(u64, 15), sumFirstFixed(&arr, 5));  // все
>     try std.testing.expectEqual(@as(u64, 1), sumFirstFixed(&arr, 1));   // только первый
> }
>
> test "sumFirstFixed: n = 0 безопасен" {
>     const arr = [_]u32{ 1, 2, 3 };
>     // Правильная версия: n=0 -> limit=0 -> пустой срез -> сумма 0
>     try std.testing.expectEqual(@as(u64, 0), sumFirstFixed(&arr, 0));
> }
>
> test "sumFirstFixed: пустой массив безопасен" {
>     const arr: []const u32 = &.{};
>     try std.testing.expectEqual(@as(u64, 0), sumFirstFixed(arr, 5));
>     try std.testing.expectEqual(@as(u64, 0), sumFirstFixed(arr, 0));
> }
>
> test "sumFirstFixed: n больше длины массива" {
>     const arr = [_]u32{ 10, 20 };
>     // @min(100, 2) = 2, суммируем весь массив
>     try std.testing.expectEqual(@as(u64, 30), sumFirstFixed(&arr, 100));
> }
>
> test "isLongerFixed: корректные результаты" {
>     try std.testing.expect(isLongerFixed("abcdef", "abc"));   // длиннее -> true
>     try std.testing.expect(!isLongerFixed("abc", "abcdef"));  // короче -> false
>     try std.testing.expect(!isLongerFixed("abc", "abc"));     // равны -> false
>     try std.testing.expect(!isLongerFixed("", "x"));          // пустая vs непустая -> false
>     try std.testing.expect(!isLongerFixed("", ""));           // обе пустые -> false
> }
>
> // ДЕМОНСТРАЦИЯ БАГА (НЕ ЗАПУСКАТЬ В ТЕСТАХ):
> // sumFirstBuggy(&arr, 0) — паника в debug:
> //   "integer overflow" при вычислении n - 1, где n = 0 (usize)
> // isLongerBuggy("abc", "abcdef") — паника в debug:
> //   "integer overflow" при вычислении 3 - 6 (usize)
>
> pub fn main() void {
>     const arr = [_]u32{ 1, 2, 3, 4, 5 };
>
>     std.debug.print("sumFirstFixed([1,2,3,4,5], 3) = {}\n", .{sumFirstFixed(&arr, 3)});
>     std.debug.print("sumFirstFixed([1,2,3,4,5], 0) = {}\n", .{sumFirstFixed(&arr, 0)});
>
>     std.debug.print("isLongerFixed(\"hello\", \"hi\")   = {}\n", .{isLongerFixed("hello", "hi")});
>     std.debug.print("isLongerFixed(\"hi\", \"hello\")   = {}\n", .{isLongerFixed("hi", "hello")});
>
>     std.debug.print("\n--- Демонстрация бага ---\n", .{});
>     std.debug.print("usize underflow: 0 - 1 = {}\n", .{@as(usize, 0) -% 1}); // wrapping sub
>     std.debug.print("(Обычный 0 - 1 для usize -> паника в debug)\n", .{});
> }
> ```

### Разбор

Это самая коварная из ловушек главы 2.2. Корень проблемы: `usize` — беззнаковый, и он **не может** стать отрицательным. Вместо `-1` получается `usize.MAX`.

В C аналогичная ловушка встречается с `strlen()` (возвращает `size_t` — беззнаковый). Именно поэтому сравнение `strlen(s) - strlen(t) > 0` опасно: если `t` длиннее `s`, результат вычитания — огромное число, а не отрицательное.

Три правила безопасной работы с `usize` в Zig:
1. Никогда не вычитать длины — сравнивать напрямую: `s.len > t.len`
2. Использовать `for (slice)` вместо индексных `while`-циклов
3. Для обратных циклов — паттерн `while (i > 0) { i -= 1; ... }`
